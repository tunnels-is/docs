# Client Config with comments
:::warning

Use the clean config below for copying, the client does not support comments in the config

:::
```json
{
    "OpenUI": false, // Enable/Disable opening the UI on client start

    // Control servers define the authentication and control endpoints
    "ControlServers": [
        {
            "ID": "tunnels", // Unique identifier for this control server
            "Host": "api.tunnels.is", // Control server hostname
            "Port": "443", // Control server port
            "CertificatePath": "", // Optional path to custom certificate for validation
            "ValidateCertificate": true // Enable/Disable certificate validation
        }
    ],

    "DisableBlockLists": false, // Completely disable DNS blocklist functionality

    // CLI specific configuration (only used in CLI mode)
    "CLIConfig": {
        "ControlServerID": "tunnels", // Which control server to use from ControlServers list
        "DeviceID": "", // Device identifier for authentication
        "ServerID": "", // Default server to connect to
        "SendStats": true, // Enable/Disable sending statistics to server
        "PinVersion": false, // Lock to a specific client version
        "SkipUpdatePrompt": false // Skip prompting for updates
    },

    // Update settings
    "AutoDownloadUpdate": true, // Automatically download updates when available
    "ExitPostUpdate": false, // Exit application after updating
    "RestartPostUpdate": false, // Restart application after updating
    "UpdateWhileConnected": false, // Allow updates while connected to VPN
    "UpdateCheckInterval": 1440, // How often to check for updates (in minutes)
    "DisableUpdates": false, // Completely disable update functionality

    // API settings for local client API
    "APIIP": "127.0.0.1", // IP address for the local API server
    "APIPort": "7777", // Port for the local API server
    "APICert": "./api.crt", // Path to API server certificate
    "APIKey": "./api.key", // Path to API server private key
    "APICertDomains": ["tunnels.app", "app.tunnels.is"], // Domains to include in API certificate
    "APICertIPs": ["127.0.0.1", "0.0.0.0"], // IP addresses to include in API certificate
    "APICertType": 0, // Certificate type (0 = self-signed, 1 = Let's Encrypt)
```
